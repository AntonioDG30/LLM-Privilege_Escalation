# File di configurazione (.env) di hackingBuddyGPT, letto a ogni run.
# Questo e' un esempio: la chiave API reale di OpenAI non va mai messa in un file
# versionato (la si tiene in ~/.bashrc lato WSL). I valori qui sotto sono quelli dei
# modelli locali via Ollama; per gpt-4o vanno cambiati llm.api_key, llm.model e llm.api_url.

# Modello LLM
llm.api_key='ollama'                   # 'ollama' per i modelli locali; per gpt-4o la chiave OpenAI
llm.model='llama3.1:8b'                # modello da usare (qui llama3.1:8b)
llm.context_size=8192                  # finestra di contesto: 8192 per i locali (limite VRAM), 64000 per gpt-4o
llm.api_url='http://localhost:11434'   # Ollama in locale; per gpt-4o: https://api.openai.com
llm.api_path='/v1/chat/completions'    # endpoint compatibile con l'API OpenAI

# Connessione SSH al bersaglio (container del benchmark)
conn.host='localhost'                  # i container girano in locale
conn.username='lowpriv'                # utente non privilegiato di partenza
conn.password='trustno1'               # password di lowpriv (credenziale del benchmark)
conn.keyfilename=''                    # nessuna chiave SSH: si usa la password

# Esecuzione e log
max_turns = 20                         # turni massimi per ogni run (il paper ne usa 60)
log_db.connection_string='/home/antonio/pt-privesc/results/run.sqlite3'   # database SQLite dei log
llm.api_timeout=900                    # timeout in secondi (alto perche' qwen3 ragiona a lungo)
